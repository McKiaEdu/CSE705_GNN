    {\LARGE \@title \par}
    \vskip 1.6em
    {\large\lineskip .5em\begin{tabular}[t]{c}\@author\end{tabular}\par}
    \vskip 1.2em
    {\large \@date \par}
  \end{center}
  \vskip 1.8em
}
